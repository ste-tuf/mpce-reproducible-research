models_nh2007 <- map2(
  .x = models_parameters_nh2007$outcomes,
  .y = models_parameters_nh2007$exposures,
  .f = \(x, y) build_model(x, y, dataset = nh2007)
)

# Extract model results
models_results_nh2007 <- map(models_nh2007, extract_model_result)
